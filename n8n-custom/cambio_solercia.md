services:
  traefik:
    image: traefik:v3.5.0
    restart: unless-stopped
    container_name: traefik
    environment:
      TZ: ${GENERIC_TIMEZONE}
    command:
      - "--api.dashboard=true"
      - "--providers.docker=true"
      - "--providers.docker.exposedbydefault=false"
      - "--entrypoints.web.address=:80"
      - "--entrypoints.web.http.redirections.entryPoint.to=websecure"
      - "--entrypoints.web.http.redirections.entryPoint.scheme=https"
      - "--entrypoints.web.http.redirections.entryPoint.permanent=true"
      - "--entrypoints.websecure.address=:443"
      - "--entrypoints.websecure.http.tls.certresolver=letsencrypt"   # ⬅️ NUEVO: resolver por defecto

      - "--certificatesresolvers.letsencrypt.acme.tlschallenge=true"
      - "--certificatesresolvers.letsencrypt.acme.email=${ACME_EMAIL}"
      - "--certificatesresolvers.letsencrypt.acme.storage=/certs/acme.json"
      # PREFETCH: pide 1 solo certificado SAN al arrancar (MAIN = www; SANS = flows, dev)
      - "--entrypoints.websecure.http.tls.domains[0].main=${SUBDOMAIN_WEB}.${DOMAIN_NAME}"
      - "--entrypoints.websecure.http.tls.domains[0].sans=${SUBDOMAIN_FLOWS}.${DOMAIN_NAME},${SUBDOMAIN_DEV}.${DOMAIN_NAME}"

      - "--log.level=DEBUG"
      - "--log.filePath=/logs/traefik.log"
      - "--accesslog=true"
      - "--accesslog.filepath=/logs/access.log"

      - "--entrypoints.web.http.middlewares=secure-stack@docker"
      - "--entrypoints.websecure.http.middlewares=secure-stack@docker"
    labels:
      - "traefik.enable=true"

      # ---------- Middlewares definidos en el proveedor Docker ----------
      # Seguridad (HSTS, XSS, NoSniff, frameDeny)
      - "traefik.http.middlewares.secure-headers.headers.stsSeconds=31536000"
      - "traefik.http.middlewares.secure-headers.headers.stsIncludeSubdomains=true"
      - "traefik.http.middlewares.secure-headers.headers.stsPreload=true"
      - "traefik.http.middlewares.secure-headers.headers.forceSTSHeader=true"
      - "traefik.http.middlewares.secure-headers.headers.browserXssFilter=true"
      - "traefik.http.middlewares.secure-headers.headers.contentTypeNosniff=true"
      - "traefik.http.middlewares.secure-headers.headers.frameDeny=true"

      # Compresión (si tu backend ya comprime, puedes omitirla)
      - "traefik.http.middlewares.compress.compress=true"

      # Rate limiting global (ajusta average/burst a tu tráfico)
      - "traefik.http.middlewares.rate-limit.ratelimit.average=50"
      - "traefik.http.middlewares.rate-limit.ratelimit.burst=100"

      # Cadena global (se aplica en ambos entrypoints vía command)
      - "traefik.http.middlewares.secure-stack.chain.middlewares=secure-headers,compress,rate-limit"

      # ---------- Routers y servicios ----------
